TABLE: PROFILES
Purpose: Master user profile
Columns: id UUID PK user_id UUID UNIQUE email TEXT full_name TEXT avatar_url TEXT college_name TEXT
department TEXT year_of_study INTEGER cgpa DECIMAL(3,2) graduation_year INTEGER linkedin_url TEXT
github_url TEXT target_role TEXT target_company TEXT placement_readiness_score INTEGER
profile_completion_percentage INTEGER created_at TIMESTAMP updated_at TIMESTAMP

TABLE: SKILLS
Purpose: Store user skills
Columns: id UUID PK user_id UUID FK skill_name TEXT skill_category TEXT skill_level INTEGER is_verified
BOOLEAN created_at TIMESTAMP
Examples: Python Java React TensorFlow Docker
Skill Levels: 1 Beginner 2 Intermediate 3 Advanced 4 Expert

TABLE: PROJECTS
Purpose: User projects
Columns: id UUID PK user_id UUID FK title TEXT description TEXT github_url TEXT live_url TEXT tech_stack
JSONB project_type TEXT difficulty_level TEXT created_at TIMESTAMP updated_at TIMESTAMP

TABLE: CERTIFICATIONS
Purpose: Store certifications
Columns: id UUID PK user_id UUID FK certificate_name TEXT issuer TEXT issue_date DATE expiry_date DATE
certificate_url TEXT created_at TIMESTAMP

TABLE: RESUMES
Purpose: Resume storage
Columns: id UUID PK user_id UUID FK file_name TEXT file_url TEXT file_size INTEGER ats_score INTEGER
resume_version TEXT analysis_status TEXT created_at TIMESTAMP

TABLE: RESUME_ANALYSIS
Purpose: Store AI resume analysis
Columns: id UUID PK resume_id UUID FK strengths JSONB weaknesses JSONB missing_keywords JSONB
recommendations JSONB overall_score INTEGER created_at TIMESTAMP

TABLE: ROLES
Purpose: Master role database
Columns: id UUID PK role_name TEXT role_description TEXT salary_min INTEGER salary_max INTEGER
industry_demand INTEGER difficulty_level TEXT created_at TIMESTAMP
Examples: AI Engineer ML Engineer Frontend Developer Backend Developer Data Scientist

TABLE: ROLE_SKILLS
Purpose: Required skills per role
Columns: id UUID PK role_id UUID FK skill_name TEXT priority TEXT weight INTEGER created_at TIMESTAMP
Priority: High Medium Low

TABLE: ROLE_MATCHES
Purpose: Store generated role matches
Columns: id UUID PK user_id UUID FK role_id UUID FK match_percentage INTEGER missing_skills JSONB
strength_skills JSONB generated_at TIMESTAMP

TABLE: COMPANIES
Purpose: Master company database

TEXT average_package INTEGER created_at TIMESTAMP

TABLE: COMPANY_ROLES

TABLE: COMPANY_REQUIREMENTS

Columns: id UUID PK company_id UUID FK minimum_cgpa DECIMAL maximum_backlogs INTEGER

TABLE: ELIGIBILITY_REPORTS
Purpose: User-company eligibility
Columns: id UUID PK user_id UUID FK company_id UUID FK eligibility_score INTEGER status TEXT
missing_skills JSONB generated_at TIMESTAMP

TABLE: SKILL_GAP_REPORTS
Purpose: Gap analysis

TABLE: ROADMAPS
